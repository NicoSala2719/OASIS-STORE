      { name Bizcochuelo Marmolado, price 9500, img  },

    } else {

  const addToCart = (item) = setCart([...cart, item]);

  const sendWhatsAppCart = () = {
    const text = `Pedido%0A${cart

    const formData = new FormData();
    formData.append(file, file);

      const res = await fetch(
        `httpsapi.cloudinary.comv1_1${CLOUD_NAME}imageupload`,

          method POST,
          body formData,
        }
      );

        alert(Error subiendo imagen);
        return;
      }

      const updated = { ...products };
      updated[category][index].img = data.secure_url;
      setProducts(updated);
    } catch {

    }
  };

                    div
                  )}

                  p className=font-semibold{item.name}p

                  {isAdmin  (
                    
                      input
                        type=number
                        value={item.price}
                        onChange={(e) = updatePrice(cat, i, e.target.value)}
                        className=border w-full mt-1
                      

                      input
                        type=file
                        accept=image
                        onChange={(e) = uploadImage(cat, i, e.target.files[0])}
                        className=mt-2
                      
                    
                  )  (
                    p${item.price}p
                  )}

                    onClick={() = addToCart(item)}
                    className=bg-blue-500 text-white w-full mt-2
